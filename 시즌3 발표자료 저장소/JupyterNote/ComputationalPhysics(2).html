def root_newton(f, df, guess, tolerance = 1.0e-6):
    """
    Uses Newton's method to find a value x near "guess"
    for which f(x) = 0, within the tolerance given.
    Default tolerance is 1.03-6, if no tolerance is 
    specified in the function call.
    
    It is require to pass this function both f(s) and f'(x).
    """
    x = guess
    dx = 2 * tolerance          # initial dx > delta
    while dx > tolerance:       # main loop -- loop until
                                # dx < tolerance
        x1 = x - f(x)/df(x)     # Point-slop form of line
        dx = abs(x - x1)        # how much have things changed?
        x = x1
        print("dx: " + str(round(dx, 3))
              + " x: " + str(round(x, 3)) \
              + " fx: " + str(round(f(x), 3)) \
              + " dfx: " + str(round(df(x), 3)))
    return x                    # Best value so far

def integrate(f_prime, a, b, m):
    dx = (b - a) / m
    s = 0.
    for i in range(0, m):
        s = s + f_prime(a + i * dx) * dx
        
    return s
